### **2.2 Stripe Integration Setup**

#### **Stripe Configuration**
```typescript
// Backend API setup (Node.js/Express example)
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: '2023-10-16',
});

// Subscription plans configuration
const SUBSCRIPTION_PLANS = {
  pro: {
    priceId: 'price_pro_monthly', // From Stripe dashboard
    features: ['ai_integration', 'cloud_sync', 'unlimited_extensions']
  },
  team: {
    priceId: 'price_team_monthly',
    features: ['collaboration', 'admin_controls', 'sso']
  }
};

// Create customer and subscription
export async function createSubscription(userId: string, email: string, priceId: string) {
  // Create Stripe customer
  const customer = await stripe.customers.create({
    email,
    metadata: { userId }
  });
  
  // Create subscription
  const subscription = await stripe.subscriptions.create({
    customer: customer.id,
    items: [{ price: priceId }],
    payment_behavior: 'default_incomplete',
    payment_settings: { save_default_payment_method: 'on_subscription' },
    expand: ['latest_invoice.payment_intent'],
  });
  
  return {
    subscriptionId: subscription.id,
    clientSecret: subscription.latest_invoice?.payment_intent?.client_secret
  };
}

// Handle extension purchases
export async function purchaseExtension(userId: string, extensionId: string, amount: number) {
  const paymentIntent = await stripe.paymentIntents.create({
    amount: amount, // in cents
    currency: 'usd',
    metadata: {
      userId,
      extensionId,
      type: 'extension_purchase'
    }
  });
  
  return { clientSecret: paymentIntent.client_secret };
}
```

### **2.3 In-App API Integration**

#### **Electron Main Process API Service**
```typescript
// src/main/services/ApiService.ts
class ApiService {
  private baseUrl = 'https://api.mothercore.dev';
  private userToken: string | null = null;
  
  async authenticate(email: string, password: string): Promise<AuthResult> {
    const response = await fetch(`${this.baseUrl}/auth/login`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, password })
    });
    
    const data = await response.json();
    if (data.token) {
      this.userToken = data.token;
      // Store securely in Electron
      this.secureStorage.setItem('user_token', data.token);
    }
    
    return data;
  }
  
  async getUserSubscription(): Promise<SubscriptionInfo> {
    return this.authenticatedRequest('/user/subscription');
  }
  
  async getExtensions(category?: string): Promise<Extension[]> {
    const params = category ? `?category=${category}` : '';
    return this.authenticatedRequest(`/extensions${params}`);
  }
  
  async purchaseExtension(extensionId: string): Promise<PurchaseResult> {
    return this.authenticatedRequest('/extensions/purchase', {
      method: 'POST',
      body: JSON.stringify({ extensionId })
    });
  }
  
  async downloadExtension(extensionId: string): Promise<Buffer> {
    const response = await this.authenticatedRequest(`/extensions/${extensionId}/download`);
    return Buffer.from(await response.arrayBuffer());
  }
  
  private async authenticatedRequest(endpoint: string, options: RequestInit = {}) {
    const response = await fetch(`${this.baseUrl}${endpoint}`, {
      ...options,
      headers: {
        'Authorization': `Bearer ${this.userToken}`,
        'Content-Type': 'application/json',
        ...options.headers
      }
    });
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.statusText}`);
    }
    
    return response.json();
  }
}
```

### **3.2 Payment Integration in App**

#### **Stripe Payment Component**
```typescript
// src/renderer/components/marketplace/PaymentModal.tsx
import { loadStripe } from '@stripe/stripe-js';
import { Elements, CardElement, useStripe, useElements } from '@stripe/react-stripe-js';

const stripePromise = loadStripe(process.env.STRIPE_PUBLISHABLE_KEY!);

const PaymentModal: React.FC<{extension: Extension, onSuccess: () => void}> = ({ extension, onSuccess }) => {
  return (
    <Elements stripe={stripePromise}>
      <PaymentForm extension={extension} onSuccess={onSuccess} />
    </Elements>
  );
};

const PaymentForm: React.FC<{extension: Extension, onSuccess: () => void}> = ({ extension, onSuccess }) => {
  const stripe = useStripe();
  const elements = useElements();
  const [processing, setProcessing] = useState(false);
  
  const handleSubmit = async (event: React.FormEvent) => {
    event.preventDefault();
    if (!stripe || !elements) return;
    
    setProcessing(true);
    
    try {
      // Create payment intent on backend
      const { clientSecret } = await window.electronAPI.createPaymentIntent({
        extensionId: extension.id,
        amount: extension.price_amount
      });
      
      // Confirm payment
      const { error, paymentIntent } = await stripe.confirmCardPayment(clientSecret, {
        payment_method: {
          card: elements.getElement(CardElement)!
        }
      });
      
      if (error) {
        throw new Error(error.message);
      }
      
      if (paymentIntent.status === 'succeeded') {
        // Download and install extension
        await window.electronAPI.installPurchasedExtension(extension.id);
        onSuccess();
      }
    } catch (error) {
      showErrorToast(error.message);
    } finally {
      setProcessing(false);
    }
  };
  
  return (
    <form onSubmit={handleSubmit} className="payment-form">
      <div className="payment-header">
        <h3>Purchase {extension.name}</h3>
        <p className="price">${extension.price_amount / 100}</p>
      </div>
      
      <div className="payment-details">
        <CardElement 
          options={{
            style: {
              base: {
                fontSize: '16px',
                color: '#ffd700',
                '::placeholder': { color: '#ffb000' }
              }
            }
          }}
        />
      </div>
      
      <div className="payment-footer">
        <button type="submit" disabled={!stripe || processing}>
          {processing ? 'Processing...' : `Pay $${extension.price_amount / 100}`}
        </button>
      </div>
    </form>
  );
};
```
